# 《悬观谜案：百年轮回》代码架构设计说明

## 一、项目概述

### 1.1 项目简介
《悬观谜案：百年轮回》是一款基于网页的交互式文字冒险游戏，采用中式恐怖题材，结合推理解密、道教机关、穿越剧情等核心玩法元素。

### 1.2 核心特性
- **颜色编码交互系统**：通过彩色高亮文字实现点击交互
- **动态叙事系统**：场景描述随时间、进度、玩家状态动态变化
- **玩家状态管理**：体力、理智、体温等多维度状态系统
- **道具与机关系统**：收集、组合道具，破解道教机关谜题
- **多结局剧情**：根据玩家选择产生不同结局

## 二、技术栈选择

### 2.1 推荐技术栈：Vue.js 3 + TypeScript

基于项目需求和开发计划，推荐使用 **Vue.js 3 + TypeScript** 技术栈：

| 类别 | 技术选型 | 选择理由 |
|------|----------|----------|
| **前端框架** | Vue.js 3 | 组件化开发、响应式系统、Composition API适合复杂状态管理 |
| **类型系统** | TypeScript | 强类型保证、IDE支持、大型项目可维护性 |
| **构建工具** | Vite | 快速开发服务器、优化的生产构建、热更新支持 |
| **状态管理** | Pinia | Vue官方推荐、TypeScript友好、模块化设计 |
| **样式方案** | Tailwind CSS + SCSS | 原子化CSS加速开发、SCSS处理复杂样式 |
| **数据存储** | localStorage + JSON | 本地存档、JSON格式游戏数据 |

### 2.2 备选技术栈对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **Vue 3 + TS** ✅ | 类型安全、生态完善、维护性好 | 学习曲线较陡 | 中大型项目、团队协作 |
| **React + TS** | 社区庞大、灵活性高 | 配置复杂、样板代码多 | 有React经验的团队 |
| **原生JS** | 无需构建、简单直接 | 难以维护、无类型检查 | 快速原型、小型项目 |
| **游戏引擎** | 功能强大、可视化编辑 | 过度设计、学习成本高 | 复杂游戏项目 |

```typescript
// src/event/EventCounter.ts
export class EventCounter {
  private counts: Map<string, number> = new Map();
  private lastTriggerTime: Map<string, number> = new Map();

  public recordEvent(eventId: string): number {
    const count = (this.counts.get(eventId) || 0) + 1;
    this.counts.set(eventId, count);
    this.lastTriggerTime.set(eventId, Date.now());
    return count;
  }

  public getEventCount(eventId: string): number {
    return this.counts.get(eventId) || 0;
  }

  public checkCooldown(eventId: string, cooldownMs: number = 5000): boolean {
    const lastTime = this.lastTriggerTime.get(eventId);
    if (!lastTime) return true;
    return Date.now() - lastTime > cooldownMs;
  }
}

// src/event/RepeatHandler.ts
export interface RepeatHandler {
  count: number;
  description: string;
  requirements?: Requirements;
  effects?: EffectData;
  variants?: RepeatVariant[];
}

export function getRepeatHandler(
  handlers: RepeatHandler[],
  count: number,
  gameState: GameState
): RepeatHandler | null {
  // 查找对应次数的处理程序
  const exactHandler = handlers.find(h => h.count === count);
  if (exactHandler) return exactHandler;
  
  // 查找最后一次的处理程序
  const lastHandler = handlers.find(h => h.count === -1);
  if (lastHandler) {
    // 检查变体
    if (lastHandler.variants) {
      return findSuitableVariant(lastHandler.variants, gameState);
    }
    return lastHandler;
  }
  
  return null;
}
```

```typescript
// src/status/StatusManager.ts
export interface StatusInfluence {
  target: keyof PlayerState;
  effect: {
    type: 'instant' | 'continuous';
    amount?: number;
    rate?: number;
  };
  condition: Condition;
}

export class StatusManager {
  private influences: StatusInfluence[] = [];
  private thresholdEvents: ThresholdEvent[] = [];

  public applyChanges(changes: Partial<PlayerState>): StatusChangeResult {
    const oldState = { ...this.getCurrentState() };
    
    // 应用变化
    Object.entries(changes).forEach(([key, value]) => {
      this.updateStatus(key as keyof PlayerState, value);
    });
    
    // 检查阈值事件
    const triggeredEvents = this.checkThresholds(oldState);
    
    // 生成变化描述
    const description = this.generateChangeDescription(oldState, changes);
    
    return {
      success: true,
      description,
      triggeredEvents
    };
  }

  public updateNaturalChanges(location: Location, time: GameTime): void {
    // 根据位置和时间计算自然状态变化
    const changes: Partial<PlayerState> = {};
    
    // 基础体力消耗
    changes.stamina = -1;
    
    // 环境影响
    if (location.isOutdoor) {
      changes.bodyTemperature = -0.2;
      if (time.weather === 'blizzard') {
        changes.bodyTemperature = -0.5;
        changes.sanity = -0.5;
      }
    }
    
    // 时间影响
    if (time.isNight) {
      changes.sanity = -1;
    }
    
    this.applyChanges(changes);
  }

  private checkThresholds(oldState: PlayerState): ThresholdEvent[] {
    const triggered: ThresholdEvent[] = [];
    const currentState = this.getCurrentState();
    
    // 理智阈值检查
    const sanityThresholds = [70, 40, 20, 10];
    sanityThresholds.forEach(threshold => {
      if (oldState.sanity > threshold && currentState.sanity <= threshold) {
        triggered.push(this.getThresholdEvent('sanity', threshold, 'below'));
      }
    });
    
    // 体温阈值检查
    const tempThresholds = [35, 32, 28];
    tempThresholds.forEach(threshold => {
      if (oldState.bodyTemperature > threshold && currentState.bodyTemperature <= threshold) {
        triggered.push(this.getThresholdEvent('temperature', threshold, 'below'));
      }
    });
    
    return triggered;
  }
}
```

```typescript
export interface Item {
  id: string;
  name: string;
  type: ItemType;
  description: string;
  detailedDescription?: string;
  usable: boolean;
  combinable: boolean;
  combineWith?: string[];
  useScenes?: string[];
  useEffect?: string;
  disposable: boolean;
  droppable: boolean;
  icon?: string;
  color?: string;
}

export type ItemType = 
  | 'key'        // 钥匙类
  | 'tool'       // 工具类
  | 'clue'       // 线索类
  | 'consumable' // 消耗品
  | 'combinable' // 组合类
  | 'quest';     // 任务类

export interface Inventory {
  maxSlots: number;
  items: string[]; // 道具ID列表
  weight: number;
  maxWeight: number;
}

export class ItemManager {
  private items: Map<string, Item> = new Map();
  private inventory: Inventory;

  public addItem(itemId: string): boolean {
    if (this.inventory.items.length >= this.inventory.maxSlots) {
      return false; // 库存已满
    }
    this.inventory.items.push(itemId);
    return true;
  }

  public useItem(itemId: string, target?: string): UseResult {
    const item = this.items.get(itemId);
    if (!item || !item.usable) {
      return { success: false, reason: '物品不可用' };
    }

    // 检查使用条件
    if (item.useScenes && !item.useScenes.includes(target || '')) {
      return { success: false, reason: '此处无法使用该物品' };
    }

    // 应用使用效果
    const result = this.applyEffect(item.useEffect);
    
    // 如果是一次性道具，从库存移除
    if (item.disposable) {
      this.removeItem(itemId);
    }

    return { success: true, result };
  }

  public combineItems(item1: string, item2: string): CombineResult {
    const itemA = this.items.get(item1);
    const itemB = this.items.get(item2);

    if (!itemA?.combinable || !itemB?.combinable) {
      return { success: false, reason: '物品无法组合' };
    }

    if (!itemA.combineWith?.includes(item2)) {
      return { success: false, reason: '这两个物品无法组合' };
    }

    // 执行组合，生成新物品
    const newItem = this.getCombinationResult(item1, item2);
    this.removeItem(item1);
    this.removeItem(item2);
    this.addItem(newItem.id);

    return { success: true, newItem };
  }
}
```

```typescript
export interface Quest {
  id: string;
  name: string;
  description: string;
  type: QuestType;
  chapter: number;
  steps: QuestStep[];
  currentStep: number;
  state: QuestState;
  prerequisites: string[];
  rewards: QuestRewards;
  hidden: boolean;
}

export interface QuestStep {
  id: string;
  description: string;
  target: string;
  action: string;
  requiredItem?: string;
  trigger?: TriggerCondition;
  completed: boolean;
}

export type QuestType = 'main' | 'side' | 'investigation' | 'puzzle' | 'collection';
export type QuestState = 'not_started' | 'active' | 'completed' | 'failed';

export class QuestManager {
  private quests: Map<string, Quest> = new Map();
  private triggers: Trigger[] = [];

  public checkTriggers(context: TriggerContext): void {
    this.triggers.forEach(trigger => {
      if (this.evaluateTrigger(trigger, context)) {
        this.activateQuest(trigger.questId);
      }
    });
  }

  public updateQuestProgress(questId: string, action: string, target: string): void {
    const quest = this.quests.get(questId);
    if (!quest || quest.state !== 'active') return;

    const currentStep = quest.steps[quest.currentStep];
    if (currentStep.action === action && currentStep.target === target) {
      this.completeStep(quest, currentStep);
    }
  }

  private completeStep(quest: Quest, step: QuestStep): void {
    step.completed = true;
    quest.currentStep++;

    if (quest.currentStep >= quest.steps.length) {
      this.completeQuest(quest);
    } else {
      this.emit('stepCompleted', { quest, step });
    }
  }
}
```

```typescript
// src/dialogue/DialogueManager.ts
export interface DialogueTree {
  npcId: string;
  initialBranch: string;
  branches: Record<string, DialogueBranch>;
}

export interface DialogueBranch {
  id: string;
  playerText: string;
  npcResponse: string;
  requirements?: Requirements;
  nextBranches?: string[];
  effects?: DialogueEffects;
}

export interface DialogueEffects {
  clueGained?: string;
  relationshipChange?: Record<string, number>;
  questTrigger?: string;
  flagSet?: string[];
}

export class DialogueManager {
  private dialogueTrees: Map<string, DialogueTree> = new Map();
  private currentDialogue: DialogueState | null = null;

  public startDialogue(npcId: string): DialogueState {
    const tree = this.dialogueTrees.get(npcId);
    if (!tree) throw new Error(`对话树不存在: ${npcId}`);

    this.currentDialogue = {
      npcId,
      currentBranch: tree.initialBranch,
      history: [],
      availableOptions: this.getAvailableOptions(tree, tree.initialBranch)
    };

    return this.currentDialogue;
  }

  public selectOption(optionIndex: number): DialogueResult {
    if (!this.currentDialogue) throw new Error('对话未开始');

    const tree = this.dialogueTrees.get(this.currentDialogue.npcId);
    const branch = tree.branches[this.currentDialogue.currentBranch];
    const selectedOption = branch.nextBranches[optionIndex];

    // 记录历史
    this.currentDialogue.history.push({
      player: branch.playerText,
      npc: branch.npcResponse
    });

    // 应用效果
    if (branch.effects) {
      this.applyDialogueEffects(branch.effects);
    }

    // 更新当前分支
    this.currentDialogue.currentBranch = selectedOption;
    this.currentDialogue.availableOptions = this.getAvailableOptions(
      tree, 
      selectedOption
    );

    return {
      npcResponse: branch.npcResponse,
      availableOptions: this.currentDialogue.availableOptions
    };
  }

  private getAvailableOptions(tree: DialogueTree, branchId: string): DialogueOption[] {
    const branch = tree.branches[branchId];
    if (!branch.nextBranches) return [];

    return branch.nextBranches
      .map(id => tree.branches[id])
      .filter(b => this.checkRequirements(b.requirements))
      .map(b => ({
        text: b.playerText,
        id: b.id
      }));
  }
}
```

```typescript
    }

    // 显示决策界面
    this.showDecisionDialog(decision);
    return true;
  }

  public makeDecision(decisionId: string, optionId: string): DecisionResult {
    const decision = this.decisions.get(decisionId);
    const option = decision.options.find(o => o.id === optionId);

    // 记录决策
    const record: DecisionRecord = {
      timestamp: new Date(),
      decisionId,
      optionId,
      outcome: this.calculateOutcome(option)
    };
    this.decisionHistory.push(record);

    // 应用即时效果
    this.applyImmediateEffects(option.immediateEffects);

    // 设置回响效果
    this.setEchoEffects(decisionId, option.longTermConsequences);

    return {
      success: true,
      outcome: record.outcome,
      consequences: option.longTermConsequences
    };
  }

  public checkEchoEffects(location: string, action: string): EchoEffect[] {
    const echoes = this.echoEffects.get(location) || [];
    return echoes.filter(echo => echo.condition(location, action));
  }
}
```

### 8.2 游戏循环实现

```typescript
class GameLoop {
  private lastUpdateTime: number = 0;
  private updateInterval: number = 1000; // 1秒更新一次

  public start(): void {
    this.lastUpdateTime = Date.now();
    this.tick();
  }

  private tick(): void {
    const currentTime = Date.now();
    const deltaTime = currentTime - this.lastUpdateTime;

    if (deltaTime >= this.updateInterval) {
      this.update();
      this.lastUpdateTime = currentTime;
    }

    requestAnimationFrame(() => this.tick());
  }

  private update(): void {
    // 更新游戏时间
    this.updateGameTime();
    
    // 更新自然状态变化
    this.statusManager.updateNaturalChanges(
      this.sceneManager.getCurrentLocation(),
      this.gameState.world.time
    );
    
    // 检查任务触发器
    this.questManager.checkTriggers(this.getTriggerContext());
    
    // 检查决策触发器
    this.decisionManager.checkTriggers(this.getTriggerContext());
    
    // 渲染更新
    this.render();
  }
}
```

### 8.3 存档系统实现

```typescript
interface SaveData {
  version: string;
  timestamp: number;
  player: PlayerState;
  world: WorldState;
  progress: ProgressState;
  flags: Record<string, boolean>;
  eventHistory: EventRecord[];
}

class SaveManager {
  private readonly SAVE_KEY = 'taosist_mystery_save';
  private readonly VERSION = '1.0.0';

  public save(slot: number = 0): void {
    const saveData: SaveData = {
      version: this.VERSION,
      timestamp: Date.now(),
      player: gameState.player,
      world: gameState.world,
      progress: gameState.progress,
      flags: gameState.flags,
      eventHistory: eventManager.getHistory()
    };

    localStorage.setItem(
      `${this.SAVE_KEY}_${slot}`,
      JSON.stringify(saveData)
    );
  }

  public load(slot: number = 0): SaveData | null {
    const data = localStorage.getItem(`${this.SAVE_KEY}_${slot}`);
    if (!data) return null;

    const saveData: SaveData = JSON.parse(data);
    
    // 版本检查
    if (saveData.version !== this.VERSION) {
      return this.migrateSaveData(saveData);
    }

    return saveData;
  }

  public autoSave(): void {
    this.save(0); // 0号槽位为自动存档
  }
}
```
