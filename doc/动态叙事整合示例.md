## 二、玩家状态管理示例

### 2.1 状态变化事件处理器

```javascript
// 状态变化处理器
class StateChangeHandler {
  constructor(gameState) {
    this.gameState = gameState;
    this.stateHistory = [];
  }

  // 应用状态变化
  applyStateChange(change) {
    const oldState = { ...this.gameState.player };

    // 更新数值
    if (change.stamina !== undefined) {
      this.gameState.player.stamina = Math.max(0, Math.min(100,
        this.gameState.player.stamina + change.stamina));
    }
    if (change.sanity !== undefined) {
      this.gameState.player.sanity = Math.max(0, Math.min(100,
        this.gameState.player.sanity + change.sanity));
    }
    if (change.temperature !== undefined) {
      this.gameState.player.temperature = Math.max(28, Math.min(38,
        this.gameState.player.temperature + change.temperature));
    }

    // 记录变化
    this.stateHistory.push({
      timestamp: this.gameState.world.time,
      change: change,
      oldState: oldState,
      newState: { ...this.gameState.player }
    });

    // 触发状态阈值事件
    this.checkThresholdEvents(oldState, this.gameState.player);

    // 返回状态变化描述
    return this.generateStateChangeDescription(oldState, this.gameState.player, change);
  }

  // 检查阈值事件
  checkThresholdEvents(oldState, newState) {
    // 理智阈值检查
    const sanityThresholds = [70, 40, 20, 10];
    sanityThresholds.forEach(threshold => {
      if (oldState.sanity > threshold && newState.sanity <= threshold) {
        this.triggerThresholdEvent('sanity', threshold, 'below');
      }
      if (oldState.sanity <= threshold && newState.sanity > threshold) {
        this.triggerThresholdEvent('sanity', threshold, 'above');
      }
    });

    // 体温阈值检查
    const tempThresholds = [35, 32, 28];
    tempThresholds.forEach(threshold => {
      if (oldState.temperature > threshold && newState.temperature <= threshold) {
        this.triggerThresholdEvent('temperature', threshold, 'below');
      }
      if (oldState.temperature <= threshold && newState.temperature > threshold) {
        this.triggerThresholdEvent('temperature', threshold, 'above');
      }
    });

    // 体力阈值检查
    const staminaThresholds = [70, 50, 30, 10];
    staminaThresholds.forEach(threshold => {
      if (oldState.stamina > threshold && newState.stamina <= threshold) {
        this.triggerThresholdEvent('stamina', threshold, 'below');
      }
    });
  }

  // 触发阈值事件
  triggerThresholdEvent(type, threshold, direction) {
    const eventId = `${type}_${threshold}_${direction}`;
    const event = thresholdEvents[eventId];

    if (event) {
      // 添加到游戏事件队列
      this.gameState.eventQueue.push({
        type: 'state_threshold',
        eventId: eventId,
        description: event.description,
        effects: event.effects
      });
    }
  }

  // 生成状态变化描述
  generateStateChangeDescription(oldState, newState, change) {
    const descriptions = [];

    // 理智变化描述
    if (change.sanity) {
      if (change.sanity < -10) {
        descriptions.push("强烈的精神冲击让你头晕目眩。");
      } else if (change.sanity < -5) {
        descriptions.push("不安感如潮水般涌来。");
      } else if (change.sanity < 0) {
        descriptions.push("你感到一丝不安。");
      } else if (change.sanity > 5) {
        descriptions.push("心神逐渐安定下来。");
      } else if (change.sanity > 0) {
        descriptions.push("心情稍微平复。");
      }
    }

    // 体温变化描述
    if (change.temperature) {
      if (change.temperature < -0.3) {
        descriptions.push("刺骨寒意瞬间穿透衣物。");
      } else if (change.temperature < -0.1) {
        descriptions.push("寒冷逐渐加剧。");
      } else if (change.temperature < 0) {
        descriptions.push("感到一丝凉意。");
      } else if (change.temperature > 0.5) {
        descriptions.push("温暖包裹全身，僵硬的手指逐渐恢复知觉。");
      } else if (change.temperature > 0.2) {
        descriptions.push("寒意稍退。");
      }
    }

    // 体力变化描述
    if (change.stamina) {
      if (change.stamina < -20) {
        descriptions.push("极度疲惫袭来，几乎站立不稳。");
      } else if (change.stamina < -10) {
        descriptions.push("疲惫感明显加重。");
      } else if (change.stamina < 0) {
        descriptions.push("感到些许疲惫。");
      } else if (change.stamina > 20) {
        descriptions.push("精力充沛，疲劳一扫而空。");
      } else if (change.stamina > 10) {
        descriptions.push("体力有所恢复。");
      }
    }

    // 组合描述
    if (descriptions.length === 0) return "";
    if (descriptions.length === 1) return descriptions[0];

    return descriptions.join(" ") + " " + this.getCombinedEffectDescription(oldState, newState);
  }

  // 获取组合效果描述
  getCombinedEffectDescription(oldState, newState) {
    const effects = [];

    // 低理智+低体温
    if (newState.sanity < 40 && newState.temperature < 35) {
      effects.push("寒冷与疯狂交织，现实开始扭曲。");
    }

    // 低体力+低理智
    if (newState.stamina < 30 && newState.sanity < 40) {
      effects.push("身心俱疲，难以集中精神。");
    }

    // 所有状态都低
    if (newState.stamina < 30 && newState.sanity < 30 && newState.temperature < 34) {
      effects.push("濒临崩溃边缘，每一步都可能是最后一步。");
    }

    return effects.join(" ");
  }
}
```

## 三、游戏流程整合示例

### 3.1 完整游戏循环

```javascript
// 游戏主循环
class GameLoop {
  constructor() {
    this.gameState = new GameState();
    this.sceneManager = new SceneManager();
    this.stateHandler = new StateChangeHandler(this.gameState);
    this.eventCounter = new EventCounter();
    this.uiRenderer = new UIRenderer();
  }

  // 主更新函数
  update() {
    // 1. 更新时间
    this.updateGameTime();

    // 2. 更新玩家状态（自然变化）
    this.updateNaturalStateChanges();

    // 3. 处理事件队列
    this.processEventQueue();

    // 4. 渲染当前场景
    this.renderCurrentScene();

    // 5. 检查任务触发器
    this.checkQuestTriggers();

    // 6. 检查决策触发器
    this.checkDecisionTriggers();
  }

  // 更新时间
  updateGameTime() {
    const oldTime = this.gameState.world.time;
    this.gameState.world.time = this.calculateNewTime(oldTime);

    // 时间变化事件
    if (this.isTimeThreshold(oldTime, this.gameState.world.time)) {
      this.triggerTimeChangeEvent(oldTime, this.gameState.world.time);
    }
  }

  // 更新自然状态变化
  updateNaturalStateChanges() {
    const location = this.sceneManager.getCurrentLocation();
    const time = this.gameState.world.time;
    const player = this.gameState.player;

    // 根据位置和时间计算状态变化
    const changes = this.calculateNaturalChanges(location, time, player);

    // 应用变化
    if (Object.keys(changes).length > 0) {
      const description = this.stateHandler.applyStateChange(changes);
      if (description) {
        this.uiRenderer.addToLog(description);
      }
    }
  }

  // 计算自然变化
  calculateNaturalChanges(location, time, player) {
    const changes = {};

    // 基础时间流逝消耗
    changes.stamina = -1; // 每分钟消耗1体力

    // 位置影响
    if (location.isOutdoor) {
      changes.temperature = -0.2; // 户外更冷
      if (this.gameState.world.weather === 'blizzard') {
        changes.temperature -= 0.3; // 暴雪中更冷
        changes.sanity = -0.5; // 暴雪影响理智
      }
    }

    // 时间影响
    if (time.isNight) {
      changes.sanity = -1; // 夜晚理智下降更快
      if (player.sanity < 50) {
        changes.sanity -= 1; // 低理智时夜晚更可怕
      }
    }

    // 状态互相影响
    if (player.temperature < 35) {
      changes.stamina -= 1; // 失温时体力消耗加快
    }
    if (player.sanity < 40) {
      changes.stamina -= 0.5; // 低理智时更容易疲惫
    }

    return changes;
  }

  // 处理玩家交互
  handleInteraction(elementId, action) {
    const location = this.sceneManager.getCurrentLocation();
    const interactive = location.getInteractive(elementId);

    if (!interactive) return;

    // 检查重复次数
    const count = this.eventCounter.recordEvent(`${location.id}_${elementId}_${action}`);
    const repeatHandler = interactive.repeat_handlers[count] ||
                         interactive.repeat_handlers[Math.min(count, 4)];

    // 检查要求
    if (repeatHandler.requirements) {
      if (!this.checkRequirements(repeatHandler.requirements)) {
        this.uiRenderer.addToLog("你现在无法这样做。");
        return;
      }
    }

    // 获取描述（如果是变体数组，随机选择一个）
    let description;
    if (Array.isArray(repeatHandler.variants)) {
      const availableVariants = repeatHandler.variants.filter(v =>
        !v.requirements || this.checkRequirements(v.requirements)
      );
      if (availableVariants.length > 0) {
        const variant = availableVariants[Math.floor(Math.random() * availableVariants.length)];
        description = variant.description;
      } else {
        description = "没有合适的反应。";
      }
    } else {
      description = repeatHandler.description;
    }

    // 显示描述
    this.uiRenderer.addToLog(description);

    // 应用效果
    if (repeatHandler.effects) {
      this.applyEffects(repeatHandler.effects);
    }

    // 触发后续事件
    if (repeatHandler.triggers) {
      this.triggerEvents(repeatHandler.triggers);
    }
  }

  // 渲染当前场景
  renderCurrentScene() {
    const locationId = this.gameState.player.location;
    const sceneData = this.sceneManager.getSceneData(locationId);

    // 生成动态描述
    const description = this.generateDynamicDescription(sceneData);

    // 渲染到UI
    this.uiRenderer.renderScene(description, sceneData.interactives);

    // 更新状态显示
    this.uiRenderer.updateStatusDisplay(this.gameState.player);
  }

  // 生成动态描述
  generateDynamicDescription(sceneData) {
    let description = sceneData.base_description;

    // 收集所有适用的变体
    const applicableVariants = [];

    // 时间变体
    const timeVariant = sceneData.variants.time[this.gameState.world.time_period];
    if (timeVariant) {
      applicableVariants.push({
        description: timeVariant.description,
        priority: timeVariant.priority || 1
      });
    }

    // 进度变体
    for (const [key, variant] of Object.entries(sceneData.variants.progress)) {
      if (this.checkRequirements(variant.requirements)) {
        applicableVariants.push({
          description: variant.description,
          priority: variant.priority || 1
        });
      }
    }

    // 状态变体
    for (const [key, variant] of Object.entries(sceneData.variants.state)) {
      if (this.checkStateRequirements(variant.requirements)) {
        applicableVariants.push({
          description: variant.description,
          priority: variant.priority || 1
        });
      }
    }

    // 按优先级排序，选择最高优先级的变体
    if (applicableVariants.length > 0) {
      applicableVariants.sort((a, b) => b.priority - a.priority);
      description = applicableVariants[0].description;

      // 如果有多个高优先级变体，可以组合它们
      const highPriorityVariants = applicableVariants.filter(v => v.priority >= 3);
      if (highPriorityVariants.length > 1) {
        description = this.combineVariants(highPriorityVariants);
      }
    }

    return description;
  }

  // 组合多个变体
  combineVariants(variants) {
    // 简单实现：选择第一个，但融入其他变体的关键元素
    let combined = variants[0].description;

    for (let i = 1; i < variants.length; i++) {
      // 提取关键句子（包含状态关键词）
      const sentences = variants[i].description.split(/[.!?]+/);
      const keySentences = sentences.filter(s =>
        s.includes('寒冷') || s.includes('理智') || s.includes('疲惫') ||
        s.includes('幻觉') || s.includes('颤抖') || s.includes('扭曲')
      );

      if (keySentences.length > 0) {
        combined += " " + keySentences.join('. ') + ".";
      }
    }

    return combined;
  }
}
```

## 五、叙事效果总结

这个动态叙事系统实现了以下效果：

### 5.1 沉浸感提升
- **环境响应**：场景描述随玩家发现、时间、状态变化
- **状态具象化**：抽象数值转化为具体生理心理体验
- **渐进恐怖**：随着理智下降，恐怖元素逐渐增强

### 5.2 防重复设计
- **智能反馈**：重复操作有不同回应，避免单调
- **状态影响**：不同状态下重复操作效果不同
- **冷却机制**：防止快速重复点击滥用

### 5.3 玩家引导
- **状态提示**：通过描述暗示玩家当前问题
- **渐进提示**：卡关时提供适度提示
- **危险预警**：状态危险时明确警告

### 5.4 叙事深度
- **多维度组合**：时间、进度、状态、事件四维度交织
- **主题一致性**：始终围绕"褪色的红与游走的墨"美学
- **逻辑严谨性**：状态变化有合理因果链

### 5.5 重玩价值
- **状态路径**：不同状态组合导致不同体验
- **选择影响**：决策改变叙事走向
- **隐藏内容**：特定状态解锁特殊内容
